Press the trigger to pull an event from the stream of time.

const triggerBtn = document.getElementById('temporal-trigger'); const displayArea = document.getElementById('event-display'); if (triggerBtn) { triggerBtn.addEventListener('click', () => { console.log("Button was clicked!"); // This helps us debug const event = historyEvents[Math.floor(Math.random() * historyEvents.length)]; const eventDate = new Date(event.year, event.month, event.day); const now = new Date(); // Calculate the difference const diff = now - eventDate; const years = Math.floor(diff / (1000 * 60 * 60 * 24 * 365.25)); displayArea.innerHTML = `

${event.title}

${years.toLocaleString()} YEARS SINCE THIS MOMENT

`; }); } else { console.log("Error: Could not find the button with ID 'temporal-trigger'"); }
